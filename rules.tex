\def\opbinintset{\{+,-,\times,/,\&,|,\opxor,\&\&,||,\ll,\gg,≤,≥,<,> \}}
\def\opbinfloatset{\{+.,-.,\times.,/.,≤.,≥.,<.,>.\}}

\def\disprule#1{
  \switch{#1}
    {Cst-Int}{ \irule{Cst-Int}
                 { }
                 { Γ ⊢ n : \tInt}
    }
    {Cst-Float}{ \irule{Cst-Float}
                   { }
                   { Γ ⊢ d : \tFloat}
    }
    {Cst-Null}{ \irule{Cst-Null}
                  { }
                  { Γ ⊢ \eNull : t*}
    }
    {Cst-Unit}{ \irule{Cst-Unit}
                  { }
                  { Γ ⊢ \eUnit : \tUnit }
    }
    {Lv-Var}{ \irule{Lv-Var}
                { x : t ∈ Γ }
                { Γ ⊢ x : t }
    }
    {Lv-Deref}{ \irule{Lv-Deref}
                  { Γ ⊢ e : t* }
                  { Γ ⊢ *e : t }
    }
    {Lv-Index}{ \irule{Lv-Index}
                  { Γ ⊢ e : \tInt \\
                    Γ ⊢ lv : t[~]
                  }
                  { Γ ⊢ lv[e] : t }
    }
    {Lv-Field}{ \irule{Lv-Field}
                  { (l, t) ∈ S \\
                    Γ ⊢ lv : S
                  }
                  { Γ ⊢ lv.l_S : t }
    }
    {Op-Int}{ \irule{Op-Int}
                { \opbin ∈ \opbinintset{} \\
                  Γ ⊢ e_1 : \tInt \\
                  Γ ⊢ e_2 : \tInt
                }{
                  Γ ⊢ e_1~\opbin~e_2 : \tInt
                }
    }
    {Op-Float}{ \irule{Op-Float}
                { \opbin ∈ \opbinfloatset{} \\
                  Γ ⊢ e_1 : \tFloat \\
                  Γ ⊢ e_2 : \tFloat
                }{
                  Γ ⊢ e_1~\opbin~e_2 : \tFloat
                }
    }
    {Eq-Num}{ \irule{Eq-Num}
                {t ∈ \{ \tInt, \tFloat \}}
                {\textsc{Eq}(t)}
    }
    {Eq-Ptr}{ \irule{Eq-Ptr}
                { }
                {\textsc{Eq}(t*)}
    }
    {Eq-Array}{ \irule{Eq-Array}
                  {\textsc{Eq}(t)}
                  {\textsc{Eq}(t[~])}
    }
    {Eq-Struct}{ \irule{Eq-Struct}
                  {∀ i ∈ [1;n]. \textsc{Eq}(t_i)}
                  {\textsc{Eq}(\tStruct{l_1 : t_1 ; … l_n : t_n})}
    }
    {Op-Eq}{ \irule{Op-Eq}
             { \opbin ∈ \{=,≠\} \\
               Γ ⊢ e_1 : t \\
               Γ ⊢ e_2 : t \\
               \textsc{Eq}(t)
             }{
               Γ ⊢ e_1~\opbin~e_2 : \tInt
             }
    }
    {Unop-Plus-Int}{ \irule{Unop-Plus-Int}
                       { Γ ⊢ e : \tInt }
                       { Γ ⊢ + e : \tInt }
    }
    {Unop-Plus-Float}{ \irule{Unop-Plus-Float}
                         { Γ ⊢ e : \tFloat }
                         { Γ ⊢ +. e : \tFloat }
    }
    {Unop-Minus-Int}{ \irule{Unop-Minus-Int}
                        { Γ ⊢ e : \tInt }
                        { Γ ⊢ - e : \tInt }
    }
    {Unop-Minus-Float}{ \irule{Unop-Minus-Float}
                          { Γ ⊢ e : \tFloat }
                          { Γ ⊢ -. e : \tFloat }
    }
    {Unop-Not}{ \irule{Unop-Not}
                { \opun ∈ \{ \sim, ! \} \\
                  Γ ⊢ e : \tInt
                }{
                  Γ ⊢ \opun~e : \tInt
                }
              }
    {Ptr-Arith}{ \irule{Ptr-Arith}
                   { \opbin ∈ \{ +_p, -_p \} \\
                     Γ ⊢ e_1 : t* \\
                     Γ ⊢ e_2 : \tInt \\
                   }
                   { Γ ⊢ e_1~\opbin~e_2 : t* }
               }
    {Addr}{ \irule{Addr}
              { Γ ⊢ lv : t }
              { Γ ⊢ \&lv : t* }
          }
    {Set}{ \irule{Set}
             { Γ ⊢ lv : t \\
               Γ ⊢ e : t
             }
             { Γ ⊢ lv ← e : t }
         }
    {Array}{ \irule{Array}
               { ∀ i ∈ [1;n] , Γ ⊢ e_i : t }
               { Γ ⊢ \eArray{ e_1 ; … ; e_n }
                       : t[~]
               }
           }
    {Struct}{ \irule{Struct}
                { ∀ i ∈ [1;n], Γ ⊢ e_i : t_i
                }
                { Γ ⊢ \eStruct{ l_1 : e_1 ; … ; l_n : e_n }
                    : \tStruct{ l_1 : t_1 ; … ; l_n : t_n }
                }
            }
    {Call}{ \irule{Call}
              { Γ ⊢ e : ( t_1 ,
                         … , t_n ) → t \\
                ∀ i ∈ [1;n], Γ ⊢ e_i : t_i
              }
              { Γ ⊢ e ( e_1
                         , …
                         , e_n
                         )
                      : t
              }
          }
    {Pass}{ \irule{Pass}
              { }
              {Γ ⊢ \iPass}
          }
    {Seq}{ \irule{Seq}
             { Γ ⊢ i_1 \\
               Γ ⊢ i_2
             }{
               Γ ⊢ i_1;i_2
             }
         }
    {Exp}{ \irule{Exp}
             { Γ ⊢ e : t }
             { Γ ⊢ e }
         }
    {Decl}{ \irule{Decl}
              { Γ ⊢ e : t
             \\ \extendLocal{Γ}{x}{t} ⊢ i
              }
              { Γ ⊢ \iDecl{x}{e}{i} }
          }
    {If}{ \irule{If}
            { Γ ⊢ e : \tInt \\
              Γ ⊢ i_1 \\
              Γ ⊢ i_2
            }
            { Γ ⊢ \iIf{e}{i_1}{i_2} }
        }
    {While}{ \irule{While}
               { Γ ⊢ e : \tInt \\
                 Γ ⊢ i
               }
               { Γ ⊢ \iWhile{e}{i} }
           }
    {Return}{ \irule{Return}
                { \vRet:t ∈ Γ
               \\ Γ ⊢ e : t
                }
                { Γ ⊢ \iReturn{e} }
            }
    {Fun}{ \irule{Fun}
             { Γ  = (Γ_G, Γ_L)
            \\ Γ' = (Γ_G,
                     [ a_1 : t_1,
                       …,
                       a_n : t_n,
                       \vRet : t
                     ]
                    )
            \\ Γ' ⊢ i
             }
             { Γ ⊢ \eFun{a_1, …, a_n}{i}
                    : (t_1, …, t_n) → t
             }
         }
    {T-Exp}{ \irule{T-Exp}
                { Γ ⊢ e : t }
                { \typh{Γ}{e}{Γ} }
            }
    {T-Var}{ \irule{T-Var}
                { Γ ⊢ e : t
               \\ Γ' = \extendGlobal{Γ}{x}{t}
                }
                { \typh{Γ}{x=e}{Γ'} }
            }
    {S-Int}{ \irule{S-Int}
               { }
               {\semtyp{m}{n}{\tInt}}
           }
    {S-Float}{ \irule{S-Float}
                 { }
                 {\semtyp{m}{d}{\tFloat}}
    }
    {S-Unit}{ \irule{S-Unit}
                { }
                {\semtyp{m}{\eUnit}{\tUnit}}
    }
    {S-Null}{ \irule{S-Null}
                { }
                {\semtyp{m}{\eNull}{τ~*}}
    }
    {S-Ptr}{ \irule{S-Ptr}
               {\semtypphi{m}{φ}{τ}}
               {\semtyp{m}{\widehat{\&}~φ}{τ~*}}
    }
    {S-Array}{ \irule{S-Array}
                 { ∀i ∈ [1;n]. \semtyp{m}{v_i}{τ}}
                 { \semtyp{m}{\widehat{ \eArray{ v_1;…;v_n } }}{τ[~]} }
    }
    {S-Struct}{ \irule{S-Struct}
                  { ∀i ∈ [1;n]. \semtyp{m}{v_i}{τ_i} }
                  { \semtyp{m}{ \widehat{ \eStruct{ l_1:v_1;…;l_n:v_n } } }
                                        { \tStruct{ l_1:τ_1;…;l_n:τ_n } }
                  }
    }
    {S-Fun}{ \irule{S-Fun}
               { }
               {\semtyp{m}{\eFun{x_1, …, x_n}{i}}{\stFun{n}}}
    }
    {AddrOf-User}{ \irule{AddrOf-User}
                 { Γ ⊢ lv : t }
                 { Γ ⊢ \eTaint{lv} : \ptrU{t} }
    }
    {GetU}{ \irule{GetU}
              { Γ ⊢ e_d : \ptrK{t}
             \\ Γ ⊢ e_s : \ptrU{t}
              }
              { Γ ⊢ \uGet{e_d}{e_s} : \tInt }
    }
    {PutU}{ \irule{PutU}
              { Γ ⊢ e_d : \ptrU{t}
             \\ Γ ⊢ e_s : \ptrK{t}
              }
              { Γ ⊢ \uPut{e_d}{e_s} : \tInt }
    }
    {AddrOf-Kernel}{ \irule{AddrOf-Kernel}
                       { Γ ⊢ lv : t }
                       { Γ ⊢ \& lv : \ptrK{t} }
    }
    {Ptr-Arith-User}{ \irule{Ptr-Arith-User}
                         { \opbin ∈ \{ +_p, -_p \} \\
                           Γ ⊢ e_1 : \ptrU{t} \\
                           Γ ⊢ e_2 : \tInt \\
                         }
                         { Γ ⊢ e_1~\opbin~e_2 : \ptrU{t} }
    }
    {END}
}

\def\semrule#1{
  \switch{#1}
    {Ctx}{ \irule{Ctx}
             { \mm{m}{i}{m'}{i'} }
             { \mm{m}{\ctxSub{C}{i}}{m'}{\ctxSub{C}{i'}} }
    }
    {Phi-Var}{ \irule{Phi-Var}
                 { a = \cLookup{x}{m} }
                 {
                   \mmlv{m}{x}{m}{a}
                 }
    }
    {Phi-Deref}{ \irule{Phi-Deref}
                   { v = \widehat{\&}~φ }
                   { \mms{*~v }{φ} }
    }
    {Phi-Struct}{ \irule{Phi-Struct}
                    {  }
                    { \mms{φ.l_S}{φ\widehat{.}l} }
    }
    {Phi-Array}{ \irule{Phi-Array}
                   { }
                   { \mms{φ[n]}{φ\widehat{[n]}} }
    }
    {Exp-Cst}{ \irule{Exp-Cst}
                 { }
                 {\mms{c}{\widehat{c}}}
             }
    {Exp-Fun}{ \irule{Exp-Fun}
                 { }
                 {\mms{f}{\widehat{f}}}
             }
    {Exp-Lv}{ \irule{Exp-Lv}
              { }
              {\mm{m}{φ}{m}{m[φ]_Φ}}
    }
    {Exp-UnOp}{ \irule{Exp-UnOp}
                { }
                {\mms{\opun~v}{\widehat{\opun}~v}}
    }
    {Exp-BinOp}{ \irule{Exp-BinOp}
                   { }
                   {\mms{v_1~\opbin~v_2}{v_1~\widehat{\opbin}~v_2}}
    }
    {Exp-AddrOf}{ \irule{Exp-AddrOf}
                    { }
                    {\mm{m}{\&~φ}{m}{\widehat{\&}~φ}}
    }
    {Exp-Set}{ \irule{Exp-Set}
                 { }
                 {\mm{m}{φ ← v}{m[φ ← v]_Φ}{v}}
    }
    {Exp-Struct}{ \irule{Exp-Struct}
                   { }
                   { \mms{          \eStruct{ l_1 : v_1 ; … ; l_n : v_n } }
                         { \widehat{\eStruct{ l_1 : v_1 ; … ; l_n : v_n } }}
                   }
    }
    {Exp-Array}{ \irule{Exp-Array}
                 { }
                 {
                   \mms{      \eArray{ v_1, … , v_n } }
                   { \widehat{\eArray{ v_1, … , v_n } }}
                 }
    }
    {Exp-Call-Return}{ \irule{Exp-Call-Return}
                          { m' = \cCleanup{m}
                         \\ v' = \cV{|m|}{v}
                          }
                          { \mm{m}{\eFun{a_1, …, a_n}{\iReturn{v}} (v_1, …, v_n)}
                               {m'}{v'}
                          }
                     }
    {Exp-Call-Ctx}{ \irule{Exp-Call-Ctx}
                        { m_1 = \cPush{m_0}{(a_1 ↦ v_1, … a_n ↦ v_n)}
                       \\ \mm{m_1}{i}{m_2}{i'}
                       \\ ∀ i ∈ [1;n], v'_i = m_2[(|m_2|, a_i)]_A
                       \\ m_3 = \cPop{m_2}
                        }
                        { \mm{m_0}{\eFun{a_1, …, a_n}{i} (v_1, …, v_n) }
                          { m_3 }{\eFun{a_1, …, a_n}{i'} (v'_1, …, v'_n) }
                        }
                  }
    {Exp-Call-Err}{ \irule{Exp-Call-Err}
                     { m' = \cPush{m}{(a_1 ↦ v_1, …, a_n ↦ v_n)}
                    \\ \msi{m'}{i} → Ω
                     }
                     { \msi{m}{\eFun{a_1, …, a_n}{i} (v_1, …, v_n) } → Ω }
                  }
    {Seq}{ \irule{Seq}
              { }
              { \mmi{m}{(\iPass;i)}{m}{i} }
    }
    {Exp}{ \irule{Exp}
             { }
             { \mmi{m}{v}{m}{\iPass} }
    }
    {Decl}{ \irule{Decl}
            { m' = \cExtend{m}{x}{v}
           \\ \mmi{m'}{i}{m''}{\iPass}
           \\ m''' = \cCleanVar{m''}{(|m|, x)}
            }
            {\mmi{m}{\iDecl{x}{v}{i}}{m'''}{\iPass}}
    }
    {Decl-Pass}{ \irule{Decl-Pass}
                     { m' = \cCleanVar{m - x}{(|m|, x)} }
                     { \mm{m}{\iDecl{x}{v}{\iPass}}
                         {m'}{\iPass}
                     }
    }
    {Decl-Return}{ \irule{Decl-Return}
                        { m' = \cCleanVar{m - x}{(|m|, x)}
                       \\ v'' = \cVV{v'}{(|m|, x)}
                        }
                        { \mm{m}{ \iDecl{x}{v}{\iReturn{v'}} }
                            {m'}{\iReturn{v''}}
                        }
    }
    {Decl-Ctx}{ \irule{Decl-Ctx}
                   { m' = \cExtend{m}{x}{v}
                   \\ \mm{m'}{i}{m''}{i'}
                   \\ v' = m''[(|m''|, x)]_A
                   \\ m''' = m'' - x
                   }
                   { \mm{m}{\iDecl{x}{v}{i}}
                       {m'''}{\iDecl{x}{v'}{i'}}
                   }
    }
    {Decl-Err}{ \irule{Decl-Err}
                    { \msi{m}{i} → Ω }
                    { \msi{m}{\iDecl{x}{v}{i}} → Ω }
    }
    {If-False}{ \irule{If-False}
                  { }
                  { \mmi{m}{\iIf{0}{i_t}{i_f}}{m}{i_f} }
    }
    {If-True}{ \irule{If-True}
                 { v ≠ 0 }
                 { \mmi{m}{\iIf{v}{i_t}{i_f}}{m}{i_t} }
    }
    {While}{ \irule{While}
               { }
               { \mmi{m}{\iWhile{e}{i}}{m}{\iThen{e}{i;\iWhile{e}{i}}}
               }
    }
    {Return}{ \irule{Return}
              { }
              { \mmi{m}{\iReturn{v};i}{m}{\iReturn{v}} }
    }
    {ET-Exp}{ \irule{ET-Exp}
                {\mm{m}{e}{m'}{v}}
                {\ph{m}{e}{m'}}
            }
    {ET-Var}{ \irule{ET-Var}
                {\mm{m}{e}{m'}{v}
               \\ m'  = (s, g)
               \\ m'' = (s, (x ↦ v)::g)
                }
                {\ph{m}{x = e}{m''}}
    }
    {T*-Nil}{ \irule{T*-Nil}
                 { }
                 {\phstar{m}{[~]}{m}}
    }
    {T*-Cons}{ \irule{T*-Cons}
                  {\ph{m}{p}{m'} \\
                   \phstar{m'}{ps}{m''}}
                  {\phstar{m}{p::ps}{m''}}
    }
    {Exp-Err}{ \irule{Exp-Err}
                 { }
                 { \msi{m}{Ω} → Ω }
    }
    {Eval-Err}{ \irule{Eval-Err}
                  { \msi{m}{e} → Ω }
                  { \msi{m}{\ctxSub{C}{e}} → Ω }
    }
    {Prog}{ \irule{Prog}
              {\phstar{([~], [~])}{P}{m}}
              {\phprog{P}{m}}
    }
    {Expr-Tainted}{ \irule{Expr-Tainted}
                      { }
                      { \mm{m}{\eTaint{φ}}{m}{\vTainted{φ}} }
    }
    {User-Get-OK}{ \irule{User-Get-OK}
                     { v  = m[φ_s]_Φ
                    \\ m' = m[φ_d ← v]_Φ
                     }
                     { \mm{m}{\uGet{φ_d}{\vTainted{φ_s}}}
                       {m'}{0}
                     }
    }
    {User-Get-Err}{ \irule{User-Get-Err}
                      {
                        ∄~φ_s. φ = \vTainted{φ_s}
                      }
                      { \mm{m}{\uGet{φ_d}{φ}}
                           {m}{-1}
                      }
    }
    {User-Put-OK}{ \irule{User-Put-OK}
                     { v = m[φ_s]_Φ
                    \\ m' = m[φ_d ← v]_Φ
                     }
                     { \mm{m}{ \uPut{\vTainted{φ_d}}{φ_s} }
                         {m'}{0}
                     }
    }
    {User-Put-Err}{ \irule{User-Put-Err}
                      {
                        ∄~φ_d. φ = \vTainted{φ_d}
                      }
                      { \mm{m}{\uPut{φ}{φ_s}}
                           {m}{-1}
                      }
    }
    {END}
}
